> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add named snapshots, autosave, manager info popovers, a manager-only view, simplified toolbar, and Pike-style identity.

**Architecture:** Backend gains snapshot management (in-memory map) and autosave file persistence (~/.grove/autosave.json). Frontend adds a ManagerView component with IC summarization, ManagerInfoPopover for org metrics, SnapshotsDropdown for save/load/branch, autosave hooks with debounced dual persistence, and the Pike-style upload screen.

**Tech Stack:** Go net/http, React, TypeScript, CSS modules, localStorage

- [ ] **Step 3: Implement snapshot methods**

```go
// internal/api/snapshots.go
package api

import (
	"fmt"
	"sort"
	"time"
)

type snapshotData struct {
	People    []Person
	Timestamp time.Time
}

func (s *OrgService) SaveSnapshot(name string) {
	s.mu.Lock()
	defer s.mu.Unlock()
	if s.snapshots == nil {
		s.snapshots = make(map[string]snapshotData)
	}
	s.snapshots[name] = snapshotData{
		People:    deepCopyPeople(s.working),
		Timestamp: time.Now(),
	}
}

func (s *OrgService) LoadSnapshot(name string) (*OrgData, error) {
	s.mu.Lock()
	defer s.mu.Unlock()
	snap, ok := s.snapshots[name]
	if !ok {
		return nil, fmt.Errorf("snapshot '%s' not found", name)
	}
	s.working = deepCopyPeople(snap.People)
	s.recycled = nil
	return &OrgData{Original: s.original, Working: s.working}, nil
}

func (s *OrgService) DeleteSnapshot(name string) {
	s.mu.Lock()
	defer s.mu.Unlock()
	delete(s.snapshots, name)
}

func (s *OrgService) ListSnapshots() []SnapshotInfo {
	s.mu.RLock()
	defer s.mu.RUnlock()
	var list []SnapshotInfo
	for name, snap := range s.snapshots {
		list = append(list, SnapshotInfo{
			Name:      name,
			Timestamp: snap.Timestamp.Format(time.RFC3339),
		})
	}
	sort.Slice(list, func(i, j int) bool {
		return list[i].Timestamp > list[j].Timestamp
	})
	return list
}
```

Add `snapshots map[string]snapshotData` to `OrgService` struct in `service.go`. Reset `s.snapshots = nil` in `Upload` (when new file uploaded).

```go
// internal/api/autosave.go
package api

import (
	"encoding/json"
	"os"
	"path/filepath"
)

func autosavePath() string {
	home, _ := os.UserHomeDir()
	return filepath.Join(home, ".grove", "autosave.json")
}

func WriteAutosave(data AutosaveData) error {
	path := autosavePath()
	os.MkdirAll(filepath.Dir(path), 0755)
	b, err := json.Marshal(data)
	if err != nil {
		return err
	}
	return os.WriteFile(path, b, 0644)
}

func ReadAutosave() (*AutosaveData, error) {
	path := autosavePath()
	b, err := os.ReadFile(path)
	if err != nil {
		if os.IsNotExist(err) {
			return nil, nil
		}
		return nil, err
	}
	var data AutosaveData
	if err := json.Unmarshal(b, &data); err != nil {
		return nil, err
	}
	return &data, nil
}

func DeleteAutosave() error {
	return os.Remove(autosavePath())
}
```
