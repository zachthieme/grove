```go
// Modified Delete
func (s *OrgService) Delete(personId string) error {
	s.mu.Lock()
	defer s.mu.Unlock()
	idx, p := s.findWorking(personId)
	if idx == -1 {
		return fmt.Errorf("person %s not found", personId)
	}
	// Unparent direct reports
	for i := range s.working {
		if s.working[i].ManagerId == personId {
			s.working[i].ManagerId = ""
		}
	}
	// Move to recycled
	s.recycled = append(s.recycled, *p)
	s.working = append(s.working[:idx], s.working[idx+1:]...)
	return nil
}

func (s *OrgService) GetRecycled() []Person {
	s.mu.RLock()
	defer s.mu.RUnlock()
	return s.recycled
}
```

```go
func (s *OrgService) Restore(personId string) error {
	s.mu.Lock()
	defer s.mu.Unlock()
	idx := -1
	for i := range s.recycled {
		if s.recycled[i].Id == personId {
			idx = i
			break
		}
	}
	if idx == -1 {
		return fmt.Errorf("person %s not found in recycle bin", personId)
	}

	person := s.recycled[idx]
	s.recycled = append(s.recycled[:idx], s.recycled[idx+1:]...)

	// Check if manager still exists in working
	if person.ManagerId != "" {
		found := false
		for _, p := range s.working {
			if p.Id == person.ManagerId {
				found = true
				break
			}
		}
		if !found {
			person.ManagerId = ""
		}
	}

	s.working = append(s.working, person)
	return nil
}

func (s *OrgService) EmptyBin() {
	s.mu.Lock()
	defer s.mu.Unlock()
	s.recycled = nil
}
```

```go
func handleDelete(svc *OrgService) http.HandlerFunc {
	return func(w http.ResponseWriter, r *http.Request) {
		var req struct {
			PersonId string `json:"personId"`
		}
		if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
			http.Error(w, "invalid JSON", http.StatusBadRequest)
			return
		}
		if err := svc.Delete(req.PersonId); err != nil {
			http.Error(w, err.Error(), http.StatusBadRequest)
			return
		}
		writeJSON(w, http.StatusOK, map[string]any{
			"working":  svc.GetWorking(),
			"recycled": svc.GetRecycled(),
		})
	}
}

	return func(w http.ResponseWriter, r *http.Request) {

}
```

```ts
// web/src/hooks/useIsManager.ts
import { useMemo } from 'react'
import type { Person } from '../api/types'

const MANAGER_PATTERN = /\b(vp|director|manager|lead|head|chief|principal)\b/i

export function isManager(person: Person, allPeople: Person[]): boolean {
  // Has direct reports?
  for (const p of allPeople) {
    if (p.managerId === person.id) return true
  }
  // Role matches manager pattern?
  return MANAGER_PATTERN.test(person.role)
}

export function useManagerSet(people: Person[]): Set<string> {
  return useMemo(() => {
    const set = new Set<string>()
    for (const p of people) {
      if (isManager(p, people)) set.add(p.id)
    }
    return set
  }, [people])
}
```

```tsx
    <div className={styles.wrapper}>
      {showActions && (
        <NodeActions
          showAdd={!!isMgr}
          onAdd={(e) => { e.stopPropagation(); onAdd?.() }}
          onDelete={(e) => { e.stopPropagation(); onDelete?.() }}
          onEdit={(e) => { e.stopPropagation(); onClick?.() }}
        />
      )}
      <div className={classNames} onClick={onClick}>
        <div className={styles.name}>{prefix}{person.name}</div>
        {showTeam && <div className={styles.team}>{person.team}</div>}
        <div className={styles.role}>{person.role || 'TBD'}</div>
      </div>
    </div>
  )
}
```
